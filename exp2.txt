#include <stdio.h>

int exists(int arr[], int size, int value) {
    for (int i = 0; i < size; i++) {
        if (arr[i] == value)
            return 1;   
    }
    return 0;   
}

int main() {
    int n, m;

    printf("Enter size of nums1: ");
    scanf("%d", &n);

    int nums1[n];
    printf("Enter elements of nums1:\n");
    for (int i = 0; i < n; i++) {
        scanf("%d", &nums1[i]);
    }

    printf("Enter size of nums2: ");
    scanf("%d", &m);

    int nums2[m];
    printf("Enter elements of nums2:\n");
    for (int i = 0; i < m; i++) {
        scanf("%d", &nums2[i]);
    }

    int answer1 = 0, answer2 = 0;

    for (int i = 0; i < n; i++) {
        if (exists(nums2, m, nums1[i])) {
            answer1++;
        }
    }

       for (int i = 0; i < m; i++) {
        if (exists(nums1, n, nums2[i])) {
            answer2++;
        }
    }

    printf("Output: [%d, %d]\n", answer1, answer2);

    return 0;
}
