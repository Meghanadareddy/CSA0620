#include <stdio.h>

// Function to count distinct elements in subarray
int countDistinct(int arr[], int start, int end) {
    int count = 0;

    for (int i = start; i <= end; i++) {
        int isDistinct = 1;

        // Check if arr[i] appeared before in this subarray
        for (int j = start; j < i; j++) {
            if (arr[i] == arr[j]) {
                isDistinct = 0;
                break;
            }
        }

        if (isDistinct)
            count++;
    }

    return count;
}

int main() {
    int n;

    printf("Enter size of array: ");
    scanf("%d", &n);

    int nums[n];

    printf("Enter elements:\n");
    for (int i = 0; i < n; i++) {
        scanf("%d", &nums[i]);
    }

    int totalSum = 0;

       for (int i = 0; i < n; i++) {
        for (int j = i; j < n; j++) {
            int distinctCount = countDistinct(nums, i, j);
            totalSum += distinctCount * distinctCount;
        }
    }

    printf("Output: %d\n", totalSum);

    return 0;
}
